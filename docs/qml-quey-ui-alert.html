Detailed Description

import QtQuick 2.4
import quey.ui 0.1


ApplicationView {
    width: Units.dp(640)
    height: Units.dp(480)

    Button {
        text: "Click me"
        anchors.centerIn: parent
        onClicked: {
            Alert.show("Alert!", "This is a very simple dialog",
                       new Alert.Button("1", function(){
                           console.log("Button 1 clicked");
                       }),
                       new Alert.Button("2", function(){
                            console.log("Button 2 clicked");
                       })
            );
        }
    }
}

import QtQuick 2.4
import quey.ui 0.1


ApplicationView {
    width: 480
    height: 640

    PageStack {
        id: pageView
        anchors.fill: parent

        Page {
            id: mainPage
            title: "Main"

            menuItems: [
                MenuItem {
                    text: "New file"
                }
            ]

            actions: [
                Action {
                    text: "New file"
                    iconName: "plus"
                    tooltip: text
                },
                Action {
                    text: "Login"
                    tooltip: text
                }
            ]

            Button {
                anchors.centerIn: parent
                text: "Settings"
                onClicked: {
                    pageView.push(settingsPage);
                }
            }
        }

        Page {
            id: settingsPage
            title: "Settings"

            CheckBox { anchors.centerIn: parent; text: "Settings" }
        }

        Component.onCompleted: {
            pageView.push(mainPage);
        }
    }
}

Detailed Description

import QtQuick 2.4
import quey.ui 0.1

ApplicationView {
    width: 640
    height: 480

    Button {
        text: "Quit"
        anchors.centerIn: parent
        onClicked: message.open()
    }

    MessageBox {
        id: message
        title: "Hey there"
        text: "Are you sure to quit?"
        buttons: [
            Button {
                text: "yes"
                onClicked: Qt.quit()
            },
            Button {
                text: "no"
                onClicked: message.close()
            }
        ]
    }

}
